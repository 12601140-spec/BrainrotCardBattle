function getCardData(id) { return CARD_DB.find(c => c.id === id); }

function generateCardHTML(card, isSleeve=false, battleData=null) {
  if(!card) return `<div class="card bg-gray-700 opacity-50 flex items-center justify-center"></div>`;
  if(isSleeve) {
    const bg = playerState.sleeveImage || defaultSleeve;
    return `<div class="card" style="background-image:url('${bg}'); background-size:cover; background-position:center;"></div>`;
  }
  let hpDisp = `HP ${card.hp}`;
  let hpColorClass = "text-red-600";
  let energyDisp = "";
  if (battleData) {
    hpDisp = `HP ${battleData.currentHp}/${card.hp}`;
    const hpPercent = battleData.currentHp / card.hp;
    hpColorClass = hpPercent > 0.5 ? 'text-green-600' : (hpPercent > 0.2 ? 'text-yellow-600' : 'text-red-600');
    let energyDots = '';
    for(let i=0; i<battleData.energy; i++) energyDots += '⭐';
    if (battleData.energy > 0) {
      energyDisp = `<div class="absolute -bottom-2 left-1/2 transform -translate-x-1/2 text-[0.6rem] bg-black/80 text-white px-2 py-0.5 rounded-full z-10 whitespace-nowrap">${energyDots}</div>`;
    }
  }
  let visualContent = '';
  if (card.image) { visualContent = card.image; } else if (card.emoji) { visualContent = `<div class="text-4xl drop-shadow-md flex items-center justify-center w-full h-full">${card.emoji}</div>`; }
  
  return `
    <div class="card type-${card.type} p-0.5 text-black flex flex-col justify-between h-full relative text-[0.4rem]">
      ${energyDisp}
      <div class="flex justify-between items-center bg-white/50 px-1 rounded rounded-b-none border-b border-gray-400">
        <span class="font-bold truncate w-[60%] leading-tight">${card.name}</span>
        <span class="font-bold ${hpColorClass} whitespace-nowrap leading-tight">${hpDisp}</span>
      </div>
      <div class="flex-1 flex justify-center items-center text-xs font-bold text-center shadow-inner bg-white/30 my-0.5 rounded overflow-hidden text-gray-800 p-1 relative">${visualContent}</div>
      <div class="bg-white/70 rounded p-0.5 flex flex-col gap-[1px] leading-tight h-[35%] overflow-hidden justify-center">
        ${card.attacks.map(atk => `<div class="flex justify-between border-b border-gray-300 pb-[1px]"><span class="font-bold truncate pr-1">⭐${atk.cost} ${atk.name}</span><span class="font-bold">${atk.dmg > 0 ? atk.dmg : '-'}</span></div>`).join('')}
      </div>
      <div class="flex justify-between bg-black/10 px-1 rounded text-gray-700 mt-0.5 leading-tight">
        <span>弱:${typeToEmoji(card.weakness)}</span><span>逃:⭐${card.retreat}</span>
      </div>
    </div>`;
}

function typeToEmoji(type) {
  const map = { fire:'🔥', water:'💧', grass:'🌿', electric:'⚡', dark:'🌌', light:'✨', normal:'⚪' }; return map[type] || '⚪';
}

  const counts = {}; playerState.collection.forEach(id => { counts[id] = (counts[id] || 0) + 1; });
  editingDeck.forEach(id => { if(counts[id]) counts[id]--; });
  const filter = document.getElementById('deck-filter').value;
  let displayCards = [];
  Object.keys(counts).forEach(idStr => {
    const id = parseInt(idStr); const count = counts[id];

let battleState = null;
function createDeck(isCpu, level=1) {
  if(!isCpu) {
    let d = [...playerState.deck]; return d.sort(() => Math.random() - 0.5).map(id => ({...getCardData(id), currentHp: getCardData(id).hp, energy: 0, uid: Math.random()}));
  } else {
    let d = []; const pool = CARD_DB.filter(c => level > 10 ? c.hp >= 80 : (level > 5 ? c.hp >= 60 : c.hp <= 80));
    for(let i=0; i<20; i++) d.push({...pool[Math.floor(Math.random()*pool.length)], currentHp: 0, energy: 0, uid: Math.random()});
    d.forEach(c => c.currentHp = c.hp); return d;
  }
}

function renderFieldCard(card, side, isBench=false) {

let selectedHandIndex = -1;
function handClick(idx) {
  if(!battleState.isPlayerTurn) return;
  selectedHandIndex = idx; const card = battleState.player.hand[idx]; const menu = document.getElementById('action-menu'); menu.innerHTML = '';
  if(battleState.phase === 'setup') {
    if(!battleState.player.active && card.stage === 0) { menu.innerHTML = `<button class="bg-blue-600 px-3 py-1 rounded text-white text-sm" onclick="playToActiveFromHand()">バトル場に出す</button>`; } else { showToast("たね(Stage0)のカードをバトル場に出してください"); return; }
  } else {
    let html = '';
    if(card.stage === 0) { if(battleState.player.bench.includes(null)) { html += `<button class="bg-green-600 px-3 py-1 rounded text-white text-sm" onclick="playToBenchFromHand()">ベンチに出す</button>`; } }
    else if(card.stage === 1) {
      const canEvolveActive = battleState.player.active && battleState.player.active.id === card.evolvesFrom;
      if(canEvolveActive) html += `<button class="bg-purple-600 px-3 py-1 rounded text-white text-sm" onclick="evolveCard('active')">バトル場を進化</button>`;
      battleState.player.bench.forEach((c, i) => { if(c && c.id === card.evolvesFrom) { html += `<button class="bg-purple-600 px-3 py-1 rounded text-white text-sm" onclick="evolveCard('bench${i}')">ベンチ${i+1}を進化</button>`; } });
    }
    html += `<button class="bg-gray-600 px-3 py-1 rounded text-white text-sm" onclick="closeActionMenu()">キャンセル</button>`; menu.innerHTML = html;
  }
  menu.classList.remove('hidden');

function playToActiveFromHand() { const card = battleState.player.hand.splice(selectedHandIndex, 1)[0]; battleState.player.active = card; closeActionMenu(); if(battleState.phase === 'setup') { cpuSetup(); battleState.phase = 'main'; startTurn(); } else { updateBattleUI(); } }
function playToBenchFromHand() { const card = battleState.player.hand.splice(selectedHandIndex, 1)[0]; const emptyIdx = battleState.player.bench.indexOf(null); battleState.player.bench[emptyIdx] = card; closeActionMenu(); updateBattleUI(); }
function evolveCard(zone) {
  const evCard = battleState.player.hand.splice(selectedHandIndex, 1)[0];
  if(zone === 'active') { evCard.currentHp = evCard.hp - (battleState.player.active.hp - battleState.player.active.currentHp); evCard.energy = battleState.player.active.energy; battleState.player.active = evCard; }
  else { const idx = parseInt(zone.replace('bench', '')); evCard.currentHp = evCard.hp - (battleState.player.bench[idx].hp - battleState.player.bench[idx].currentHp); evCard.energy = battleState.player.bench[idx].energy; battleState.player.bench[idx] = evCard; }
  createEffect('✨', document.getElementById(`player-${zone}`)); closeActionMenu(); updateBattleUI();
}

function selectZone(zone) {
  if(!battleState.isPlayerTurn || battleState.phase === 'setup') return;
  const targetCard = zone === 'active' ? battleState.player.active : battleState.player.bench[parseInt(zone.replace('bench',''))]; if(!targetCard) return;
  const menu = document.getElementById('action-menu'); let html = '';
  if(battleState.player.energyZone > 0 && !battleState.player.energyAttachedThisTurn) { html += `<button class="bg-yellow-500 px-3 py-1 rounded text-black font-bold text-sm" onclick="attachEnergy('${zone}')">エネをつける</button>`; }
  if(zone === 'active') { targetCard.attacks.forEach((atk, i) => { const canUse = targetCard.energy >= atk.cost; html += `<button class="${canUse ? 'bg-red-600' : 'bg-gray-600'} px-3 py-1 rounded text-white text-sm" ${canUse?'':'disabled'} onclick="useAttack(${i})">${atk.name} (⭐${atk.cost})</button>`; }); }
  html += `<button class="bg-gray-600 px-3 py-1 rounded text-white text-sm" onclick="closeActionMenu()">閉じる</button>`; menu.innerHTML = html; menu.classList.remove('hidden');
}

function attachEnergy(zone) { battleState.player.energyZone--; battleState.player.energyAttachedThisTurn = true; if(zone === 'active') battleState.player.active.energy++; else battleState.player.bench[parseInt(zone.replace('bench',''))].energy++; createEffect('⭐', document.getElementById(`player-${zone}`)); closeActionMenu(); updateBattleUI(); }

function startTurn() { const p = battleState.isPlayerTurn ? battleState.player : battleState.cpu; drawCard(battleState.isPlayerTurn ? 'player' : 'cpu'); p.energyZone++; p.energyAttachedThisTurn = false; updateBattleUI(); if(!battleState.isPlayerTurn) setTimeout(cpuTurn, 1000); }

function actionRetreat() {
  const p = battleState.player; if (!p.active) return;
  if(p.active.energy < p.active.retreat) { showToast(`逃げるにはエネルギーが ${p.active.retreat} 個必要です！`); return; }
  const benchIdx = p.bench.findIndex(c => c !== null); if(benchIdx === -1) { showToast("交代できるベンチがいません！"); return; }
  p.active.energy -= p.active.retreat; const temp = p.active; p.active = p.bench[benchIdx]; p.bench[benchIdx] = temp; showToast(`${temp.name} は逃げた！`); updateBattleUI();
}

  const attackerEl = document.getElementById(`${side}-active`); const defenderSide = side === 'player' ? 'cpu' : 'player'; const defenderEl = document.getElementById(`${defenderSide}-active`);
  attackerEl.classList.add(side==='player' ? 'anim-attack' : 'anim-attack-cpu'); setTimeout(() => attackerEl.classList.remove('anim-attack', 'anim-attack-cpu'), 300);
  setTimeout(() => {
    let damage = atk.dmg; if(damage > 0 && defender.weakness === attacker.type) damage += 20;
    if(damage > 0) { defenderEl.classList.add('anim-damage', 'anim-shake'); setTimeout(() => defenderEl.classList.remove('anim-damage', 'anim-shake'), 400); createHitEffect(defenderEl, damage); defender.currentHp -= damage; }
    if(atk.effect === 'heal10') attacker.currentHp = Math.min(attacker.hp, attacker.currentHp + 10); if(atk.effect === 'heal20') attacker.currentHp = Math.min(attacker.hp, attacker.currentHp + 20); if(atk.effect === 'heal30') attacker.currentHp = Math.min(attacker.hp, attacker.currentHp + 30); if(atk.effect === 'self10') attacker.currentHp -= 10; if(atk.effect === 'discard1' && battleState[side].energyZone > 0) battleState[side].energyZone--;
    updateBattleUI(); setTimeout(() => { checkFaint(defenderSide, defender); }, 500);
  }, 150);
}

function checkFaint(side, card) {
  if(card.currentHp <= 0) {
    createEffect('💥', document.getElementById(`${side}-active`), 3); battleState[side].active = null; const winnerSide = side === 'player' ? 'cpu' : 'player'; battleState[winnerSide].points++; updateBattleUI();
    if(battleState[winnerSide].points >= 3) { endGame(winnerSide); return; }
    setTimeout(() => { const benchIdx = battleState[side].bench.findIndex(c => c !== null); if(benchIdx !== -1) { battleState[side].active = battleState[side].bench[benchIdx]; battleState[side].bench[benchIdx] = null; updateBattleUI(); proceedTurn(winnerSide); } else { endGame(winnerSide); } }, 1000);
  } else { proceedTurn(side === 'player' ? 'cpu' : 'player'); }
}
function proceedTurn(currentSide) { battleState.isPlayerTurn = currentSide === 'cpu'; battleState.turn++; startTurn(); }

async function delay(ms) { return new Promise(r => setTimeout(r, ms)); }
async function cpuSetup() { const c = battleState.cpu; const basicIdx = c.hand.findIndex(card => card.stage === 0); if(basicIdx !== -1) { c.active = c.hand.splice(basicIdx, 1)[0]; } else { const deckIdx = c.deck.findIndex(card => card.stage === 0); if(deckIdx !== -1) c.active = c.deck.splice(deckIdx, 1)[0]; } }
async function cpuTurn() {
  const c = battleState.cpu; const p = battleState.player; if(battleState.phase === 'setup') return; await delay(1000);
  for(let i=c.hand.length-1; i>=0; i--) { if(c.hand[i].stage === 0 && c.bench.includes(null)) { const emptyIdx = c.bench.indexOf(null); c.bench[emptyIdx] = c.hand.splice(i, 1)[0]; updateBattleUI(); await delay(500); } }
  if(selectedCpuLevel >= 5) {
    for(let i=c.hand.length-1; i>=0; i--) {
      const card = c.hand[i];
      if(card.stage === 1) {
        if(c.active && c.active.id === card.evolvesFrom) { card.currentHp = card.hp - (c.active.hp - c.active.currentHp); card.energy = c.active.energy; c.active = c.hand.splice(i, 1)[0]; createEffect('✨', document.getElementById('cpu-active')); updateBattleUI(); await delay(500); continue; }
        for(let j=0; j<3; j++) { if(c.bench[j] && c.bench[j].id === card.evolvesFrom) { card.currentHp = card.hp - (c.bench[j].hp - c.bench[j].currentHp); card.energy = c.bench[j].energy; c.bench[j] = c.hand.splice(i, 1)[0]; createEffect('✨', document.getElementById(`cpu-bench`)); updateBattleUI(); await delay(500); break; } }
      }
    }
  }
  if(c.energyZone > 0 && c.active && selectedCpuLevel >= 1) { c.active.energy++; c.energyZone--; createEffect('⭐', document.getElementById('cpu-active')); updateBattleUI(); await delay(500); }
  if(c.active) {
    let bestAtkIdx = -1; let maxDmg = -1; c.active.attacks.forEach((atk, i) => { if(c.active.energy >= atk.cost && atk.dmg > maxDmg) { maxDmg = atk.dmg; bestAtkIdx = i; } });
    if(bestAtkIdx !== -1) { executeAttack('cpu', c.active, p.active, c.active.attacks[bestAtkIdx]); return; }
  }
  battleState.isPlayerTurn = true; battleState.turn++; startTurn();
}

function createEffect(emoji, targetEl, scale=1) { if(!targetEl) return; const rect = targetEl.getBoundingClientRect(); const el = document.createElement('div'); el.className = 'hit-effect'; el.innerText = emoji; el.style.left = `${rect.left + rect.width/2}px`; el.style.top = `${rect.top + rect.height/2}px`; el.style.fontSize = `${4 * scale}rem`; document.getElementById('battle-effect-layer').appendChild(el); setTimeout(() => el.remove(), 500); }
function createHitEffect(targetEl, damage) { createEffect('💥', targetEl); const rect = targetEl.getBoundingClientRect(); const dmgEl = document.createElement('div'); dmgEl.className = 'absolute font-black text-red-500 text-5xl text-glow pointer-events-none z-50'; dmgEl.innerText = `-${damage}`; dmgEl.style.left = `${rect.left + rect.width/2 - 20}px`; dmgEl.style.top = `${rect.top + rect.height/2 - 20}px`; dmgEl.style.animation = 'attack-slide 1s ease-out forwards, blast 1s forwards'; document.getElementById('battle-effect-layer').appendChild(dmgEl); setTimeout(() => dmgEl.remove(), 1000); }
function endGame(winnerSide) {
  const resScreen = document.getElementById('battle-result'); resScreen.classList.remove('hidden'); resScreen.classList.add('flex');
  if(winnerSide === 'player') { document.getElementById('result-title').innerText = "VICTORY!"; document.getElementById('result-title').className = "text-6xl font-black mb-4 text-blue-400 text-glow"; const reward = selectedCpuLevel * 10; document.getElementById('result-reward').innerText = `+ ${reward} 🪙`; playerState.coins += reward; saveData(); }
  else { document.getElementById('result-title').innerText = "DEFEAT..."; document.getElementById('result-title').className = "text-5xl font-black mb-4 text-red-500 text-center leading-tight"; document.getElementById('result-reward').innerText = `+ 10 🪙`; playerState.coins += 10; saveData(); }
}
function endBattle() { battleState = null; updateUI(); nav('home'); }
